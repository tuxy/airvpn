1//! AirVPN authentication protocol implementation.
2//!
3//! This module implements the proprietary encrypted request/response protocol
4//! used by AirVPN's API. The protocol works as follows:
5//!
6//! 1. Generate a random AES-256 key and IV.
7//! 2. Encrypt the key+IV with the server's RSA public key (PKCS#1 v1.5) → `s`.
8//! 3. Encrypt the request parameters with AES-256-CBC (PKCS7-padded) → `d`.
9//! 4. POST `s` and `d` (base64-encoded, URL-encoded) to a bootstrap server.
10//! 5. Decrypt the response with the same AES key and IV.
11//!
12//! This module exposes low-level helpers ([`encrypt_request`], [`decrypt_response`],
13//! [`build_post_body`]) as well as the mid-level [`make_rsa_pubkey`] for key
14//! construction. In most cases [`AirVPN::builder`] should be used instead
15//! (Even I don't consider the low-level helpers very easy to grasp).
16//! 
17//!
18//! [`AirVPN::builder`]: crate::AirVPN::builder
19

29/// The result of encrypting a request with the AirVPN protocol.
30///
31/// Contains the RSA-encrypted key/IV blob (`s`), the AES-encrypted data blob
32/// (`d`), and the raw AES key and IV needed to decrypt the server's response.
33pub struct EncryptedRequest {
34    /// RSA-encrypted blob containing the AES key and IV (base64-encoded before POST).
35    pub s: Vec<u8>,
36    /// AES-256-CBC encrypted request parameters (base64-encoded before POST).
37    pub d: Vec<u8>,
38    /// The raw 32-byte AES-256 key. Keep this to decrypt the response.
39    pub aes_key: Vec<u8>,
40    /// The raw 16-byte AES-CBC initialization vector. Keep this to decrypt the response.
41    pub aes_iv: Vec<u8>,
42}
43
44/// Encode a list of key-value byte pairs as a newline-delimited `base64(key):base64(value)` string.
45///
46/// This is the wire format the AirVPN API expects for both the RSA-encrypted
47/// envelope (`s`) and the AES-encrypted payload (`d`).
48pub fn encode_assoc(params: &[(Vec<u8>, Vec<u8>)]) -> String {
49    params
50        .iter()
51        .map(|(key, value)| {
52            let key_b64 = BASE64.encode(key);
53            let val_b64 = BASE64.encode(value);
54            format!("{}:{}", key_b64, val_b64)
55        })
56        .collect::<Vec<_>>()
57        .join("\n")
58}
59
60/// Convenience wrapper around [`encode_assoc`] that accepts string pairs.
61///
62/// Each string is converted to its UTF-8 byte representation before encoding.
63pub fn encode_assoc_str(params: &[(String, String)]) -> String {
64    let byte_params: Vec<(Vec<u8>, Vec<u8>)> = params
65        .iter()
66        .map(|(k, v)| (k.as_bytes().to_vec(), v.as_bytes().to_vec()))
67        .collect();
68    encode_assoc(&byte_params)
69}

71/// Construct an RSA public key from base64-encoded modulus and exponent.
72///
73/// The modulus and exponent are the values found in the AirVPN manifest
74/// (attributes `auth_rsa_modulus` and `auth_rsa_exponent`), or the
75/// defaults in [`DEFAULT_RSA_MODULUS`]/[`DEFAULT_RSA_EXPONENT`].
76///
77/// [`DEFAULT_RSA_MODULUS`]: crate::constants::DEFAULT_RSA_MODULUS
78/// [`DEFAULT_RSA_EXPONENT`]: crate::constants::DEFAULT_RSA_EXPONENT
79pub fn make_rsa_pubkey(modulus_b64: &str, exponent_b64: &str) -> Result<RsaPublicKey> {
80    let modulus_bytes = BASE64.decode(modulus_b64)?;
81    let exponent_bytes = BASE64.decode(exponent_b64)?;
82    let n = BigUint::from_bytes_be(&modulus_bytes);
83    let e = BigUint::from_bytes_be(&exponent_bytes);
84    RsaPublicKey::new(n, e).map_err(|e| Error::InvalidKey(e.to_string()))
85}

87/// Encrypt request parameters using the AirVPN protocol.
88///
89/// Returns an [`EncryptedRequest`] containing the two blobs (`s` and `d`)
90/// to POST, along with the AES key/IV needed to decrypt the response.
91///
92/// # Arguments
93///
94/// * `parameters` — Key-value pairs to send (e.g. `act`, `login`, `password`).
95/// * `rsa_modulus_b64` — Base64-encoded RSA modulus (from manifest or defaults).
96/// * `rsa_exponent_b64` — Base64-encoded RSA exponent (from manifest or defaults).
97pub fn encrypt_request(
98    parameters: &[(String, String)],
99    rsa_modulus_b64: &str,
100    rsa_exponent_b64: &str,
101) -> Result<EncryptedRequest> {
102    let aes_key = new_aes_key();
103    let aes_iv = new_aes_iv();
104
105    let rsa_pubkey = make_rsa_pubkey(rsa_modulus_b64, rsa_exponent_b64)?;
106
107    let param_s = encode_assoc(&[
108        (b"key".to_vec(), aes_key.clone()),
109        (b"iv".to_vec(), aes_iv.clone()),
110    ]);
111    let encrypted_s = rsa_pubkey
112        .encrypt(&mut OsRng, Pkcs1v15Encrypt, param_s.as_bytes())
113        .map_err(|e| Error::Encryption(e.to_string()))?;
114
115    let data = encode_assoc_str(parameters);
116    let padded = pkcs7_pad(data.as_bytes());
117    let encrypted_d = aes_cbc_encrypt(&aes_key, &aes_iv, &padded);
118
119    Ok(EncryptedRequest {
120        s: encrypted_s,
121        d: encrypted_d,
122        aes_key,
123        aes_iv,
124    })
125}

127/// Decrypt a response body received from an AirVPN bootstrap server.
128///
129/// `encrypted_data` is the raw bytes from the HTTP response body.
130/// `aes_key` and `aes_iv` are from the [`EncryptedRequest`] used to make
131/// the request.
132pub fn decrypt_response(
133    encrypted_data: &[u8],
134    aes_key: &[u8],
135    aes_iv: &[u8],
136) -> Result<Vec<u8>> {
137    let decrypted = aes_cbc_decrypt(aes_key, aes_iv, encrypted_data);
138    pkcs7_unpad(&decrypted)
139}

141/// Build a URL-encoded POST body from the two encrypted blobs.
142///
143/// The result is suitable for use as the body of a `POST` request with
144/// `Content-Type: application/x-www-form-urlencoded`.
145pub fn build_post_body(encrypted_s: &[u8], encrypted_d: &[u8]) -> String {
146    let s_encoded = BASE64.encode(encrypted_s);
147    let d_encoded = BASE64.encode(encrypted_d);
148    form_urlencoded::Serializer::new(String::new())
149        .append_pair("s", &s_encoded)
150        .append_pair("d", &d_encoded)
151        .finish()
152}

154/// Apply PKCS#7 padding to `data`, adding between 1 and 16 bytes
155/// so the result length is a multiple of the AES block size (16).
156fn pkcs7_pad(data: &[u8]) -> Vec<u8> {
157    let padding_len = 16 - (data.len() % 16);
158    let mut padded = data.to_vec();
159    padded.extend(std::iter::repeat(padding_len as u8).take(padding_len));
160    padded
161}
162
163/// Remove PKCS#7 padding. Returns an error if the padding is invalid.
164fn pkcs7_unpad(data: &[u8]) -> Result<Vec<u8>> {
165    if data.is_empty() {
166        return Err(Error::Decryption("empty data".into()));
167    }
168    let padding_len = *data.last().unwrap() as usize;
169    if padding_len == 0 || padding_len > 16 || padding_len > data.len() {
170        return Err(Error::Decryption("invalid PKCS7 padding".into()));
171    }
172    for &byte in &data[data.len() - padding_len..] {
173        if byte != padding_len as u8 {
174            return Err(Error::Decryption("invalid PKCS7 padding".into()));
175        }
176    }
177    Ok(data[..data.len() - padding_len].to_vec())
178}

180/// Encrypt `data` with AES-256-CBC using the given 32-byte `key` and 16-byte `iv`.
181fn aes_cbc_encrypt(key: &[u8], iv: &[u8], data: &[u8]) -> Vec<u8> {
182    let cipher = Aes256::new_from_slice(key).expect("invalid AES-256 key");
183    let mut ciphertext = Vec::with_capacity(data.len());
184    let mut prev = iv.to_vec();
185
186    for chunk in data.chunks(16) {
187        let mut block = [0u8; 16];
188        block[..chunk.len()].copy_from_slice(chunk);
189        for i in 0..16 {
190            block[i] ^= prev[i];
191        }
192        let mut ga = GenericArray::clone_from_slice(&block);
193        cipher.encrypt_block(&mut ga);
194        prev.copy_from_slice(&ga);
195        ciphertext.extend_from_slice(&ga);
196    }
197
198    ciphertext
199}

201/// Decrypt `data` with AES-256-CBC using the given 32-byte `key` and 16-byte `iv`.
202///
203/// **Note:** This does *not* remove padding. Call [`pkcs7_unpad`] on the result
204/// if the input was PKCS7-padded.
205fn aes_cbc_decrypt(key: &[u8], iv: &[u8], data: &[u8]) -> Vec<u8> {
206    let cipher = Aes256::new_from_slice(key).expect("invalid AES-256 key");
207    let mut plaintext = Vec::with_capacity(data.len());
208    let mut prev = iv.to_vec();
209
210    for chunk in data.chunks(16) {
211        let mut ga = GenericArray::clone_from_slice(chunk);
212        let encrypted = ga.clone();
213        cipher.decrypt_block(&mut ga);
214        let block: Vec<u8> = ga.to_vec();
215        let mut decrypted = [0u8; 16];
216        for i in 0..16 {
217            decrypted[i] = block[i] ^ prev[i];
218        }
219        prev = encrypted.to_vec();
220        plaintext.extend_from_slice(&decrypted);
221    }
222
223    plaintext
224}

226/// Generate a random 32-byte AES-256 key.
227fn new_aes_key() -> Vec<u8> {
228    let mut key = [0u8; 32];
229    OsRng.fill_bytes(&mut key);
230    key.to_vec()
231}
232
233/// Generate a random 16-byte AES-CBC IV.
234fn new_aes_iv() -> Vec<u8> {
235    let mut iv = [0u8; 16];
236    OsRng.fill_bytes(&mut iv);
237    iv.to_vec()
238}
