1//! High-level AirVPN client with a builder-pattern API.
2//!
3//! The entry point is [`AirVPN::builder()`], which returns an [`AirVPNBuilder`].
4//! Configure credentials, server, key name, port, etc. on the builder, then
5//! call [`AirVPNBuilder::build`] to authenticate, fetch the profile, and
6//! produce a complete [`WireGuardConfig`].
7//!
8//! For the public status API (no auth required), use [`AirVPN::fetch_status`].
9//!
10//! # Example
11//!
12//! ```no_run
13//! use airvpn::AirVPN;
14//!
15//! # async fn example() -> Result<(), Box<dyn std::error::Error>> {
16//! let config = AirVPN::builder()
17//!     .credentials("user", "pass")
18//!     .server("Algorab")
19//!     .build()
20//!     .await?;
21//!
22//! println!("{}", config.to_conf());
23//! # Ok(())
24//! # }
25//! ```

37/// Builder for configuring and executing an AirVPN authentication + config fetch.
38///
39/// Obtain one via [`AirVPN::builder()`], then chain setter methods and finish
40/// with [`build`](AirVPNBuilder::build).
41///
42/// # Defaults
43///
44/// | Field | Default |
45/// |-------|---------|
46/// | `key_name` | `"Default"` |
47/// | `port` | `51820` |
48/// | `bootstrap_urls` | [`DEFAULT_BOOTSTRAP_URLS`] |
49/// | `timeout` | 15 seconds |
50/// | RSA modulus/exponent | [`DEFAULT_RSA_MODULUS`] / [`DEFAULT_RSA_EXPONENT`] |
51///
52/// [`DEFAULT_BOOTSTRAP_URLS`]: crate::constants::DEFAULT_BOOTSTRAP_URLS
53/// [`DEFAULT_RSA_MODULUS`]: crate::constants::DEFAULT_RSA_MODULUS
54/// [`DEFAULT_RSA_EXPONENT`]: crate::constants::DEFAULT_RSA_EXPONENT
55#[derive(Clone)]
56pub struct AirVPNBuilder {
57    username: Option<String>,
58    password: Option<String>,
59    server: Option<String>,
60    key_name: String,
61    port: u16,
62    bootstrap_urls: Option<Vec<String>>,
63    timeout: Duration,
64    rsa_modulus: Option<String>,
65    rsa_exponent: Option<String>,
66}
67
68impl AirVPNBuilder {
69    /// Set the AirVPN username and password.
70    ///
71    /// Required. Calling [`build`](AirVPNBuilder::build) without credentials
72    /// will return [`Error::MissingCredentials`].
73    pub fn credentials(mut self, username: &str, password: &str) -> Self {
74        self.username = Some(username.to_string());
75        self.password = Some(password.to_string());
76        self
77    }
78
79    /// Set the target server name (e.g. `"Algorab"`).
80    ///
81    /// If omitted, the first server in the manifest is used.
82    pub fn server(mut self, server: &str) -> Self {
83        self.server = Some(server.to_string());
84        self
85    }
86
87    /// Set the WireGuard key profile name (defaults to `"Default"`).
88    ///
89    /// AirVPN supports multiple key profiles per account. This selects which
90    /// one to use when extracting the WireGuard configuration.
91    pub fn key_name(mut self, key_name: &str) -> Self {
92        self.key_name = key_name.to_string();
93        self
94    }
95
96    /// Set the WireGuard endpoint port (defaults to `51820`).
97    pub fn port(mut self, port: u16) -> Self {
98        self.port = port;
99        self
100    }
101
102    /// Override the default bootstrap server URLs.
103    ///
104    /// The client will try each URL in order until one responds. After the
105    /// manifest is fetched, the URL list is updated from the server.
106    pub fn bootstrap_urls(mut self, urls: Vec<String>) -> Self {
107        self.bootstrap_urls = Some(urls);
108        self
109    }
110
111    /// Set the HTTP request timeout (defaults to 15 seconds).
112    pub fn timeout(mut self, timeout: Duration) -> Self {
113        self.timeout = timeout;
114        self
115    }
116
117    /// Authenticate with AirVPN, fetch the manifest and user profile, and
118    /// creates a [`WireGuardConfig`].
119    ///
120    /// This process:
121    ///
122    /// 1. Encrypts a `manifest` request with the current AirVPN RSA/AES keys.
123    /// 2. Tries each bootstrap URL until one succeeds.
124    /// 3. Decrypts and parses the XML manifest.
125    /// 4. Updates RSA keys and bootstrap URLs from the manifest.
126    /// 5. Repeats with a `user` request using updated keys.
127    /// 6. Extracts WireGuard config from the user profile.
128    /// 7. Resolves the server endpoint from the manifest.
129    pub async fn build(self) -> Result<WireGuardConfig> {
130        let username = self.username.ok_or(Error::MissingCredentials)?;
131        let password = self.password.ok_or(Error::MissingCredentials)?;
132
133        let rsa_modulus = self
134            .rsa_modulus
135            .unwrap_or_else(|| DEFAULT_RSA_MODULUS.to_string());
136        let rsa_exponent = self
137            .rsa_exponent
138            .unwrap_or_else(|| DEFAULT_RSA_EXPONENT.to_string());
139        let bootstrap_urls = self
140            .bootstrap_urls
141            .unwrap_or_else(|| DEFAULT_BOOTSTRAP_URLS.iter().map(|s| s.to_string()).collect());
142
143        let http_client = reqwest::Client::builder()
144            .timeout(self.timeout)
145            .build()?;
146
147        let manifest_text = fetch(
148            &http_client,
149            "manifest",
150            &username,
151            &password,
152            &rsa_modulus,
153            &rsa_exponent,
154            &bootstrap_urls,
155        )
156        .await?;
157        let manifest_doc = roxmltree::Document::parse(&manifest_text)?;
158
159        let root = manifest_doc.root_element();
160        if let Some(error) = root.attribute("error") {
161            if !error.is_empty() {
162                return Err(Error::Server(error.to_string()));
163            }
164        }
165
166        let mut updated_modulus = rsa_modulus;
167        let mut updated_exponent = rsa_exponent;
168        if let Some(modulus) = root.attribute("auth_rsa_modulus") {
169            if !modulus.is_empty() {
170                updated_modulus = modulus.to_string();
171            }
172        }
173        if let Some(exponent) = root.attribute("auth_rsa_exponent") {
174            if !exponent.is_empty() {
175                updated_exponent = exponent.to_string();
176            }
177        }
178
179        let mut updated_urls = bootstrap_urls;
180        let urls_node = root.descendants().find(|n| n.has_tag_name("urls"));
181        if let Some(urls_node) = urls_node {
182            let mut new_urls: Vec<String> = Vec::new();
183            for url_node in urls_node.descendants().filter(|n| n.has_tag_name("url")) {
184                if let Some(addr) = url_node.attribute("address") {
185                    let addr = addr.trim().to_string();
186                    if !addr.is_empty() {
187                        new_urls.push(addr);
188                    }
189                }
190            }
191            if !new_urls.is_empty() {
192                let old_urls: Vec<String> = updated_urls
193                    .into_iter()
194                    .filter(|u| !new_urls.contains(u))
195                    .collect();
196                new_urls.extend(old_urls);
197                updated_urls = new_urls;
198            }
199        }
200
201        let user_text = fetch(
202            &http_client,
203            "user",
204            &username,
205            &password,
206            &updated_modulus,
207            &updated_exponent,
208            &updated_urls,
209        )
210        .await?;
211        let user_doc = roxmltree::Document::parse(&user_text)?;
212
213        let user_root = user_doc.root_element();
214        let user_node = if user_root.has_tag_name("user") {
215            user_root
216        } else {
217            user_root
218                .descendants()
219                .find(|n| n.has_tag_name("user"))
220                .ok_or_else(|| Error::Wireguard("no <user> node in response".into()))?
221        };
222
223        if let Some(error) = user_node.attribute("error") {
224            if !error.is_empty() {
225                return Err(Error::Server(error.to_string()));
226            }
227        }
228
229        let mut config = wireguard::extract_wg_config(user_node, &self.key_name)?;
230        let endpoint = wireguard::resolve_endpoint(root, self.server.as_deref());
231        config.endpoint = endpoint;
232        config.port = self.port;
233
234        Ok(config)
235    }
236}

254/// Entry point for the AirVPN library.
255///
256/// Refer to [`AirVPN::builder()`] for building an authenticated request, or
257/// [`AirVPN::fetch_status()`] for the public status API.

267    ///

274/// Encrypt and POST an API request, trying each bootstrap URL.
275async fn fetch(
276    http_client: &reqwest::Client,
277    action: &str,
278    login: &str,
279    password: &str,
280    rsa_modulus: &str,
281    rsa_exponent: &str,
282    bootstrap_urls: &[String],
283) -> Result<String> {
284    let parameters: Vec<(String, String)> = vec![
285        ("act".into(), action.into()),
286        ("login".into(), login.into()),
287        ("password".into(), password.into()),
288        ("software".into(), format!("EddieCLI_{}", DOCUMENT_VERSION)),
289        ("system".into(), "linux".into()),
290        ("arch".into(), "x86_64".into()),
291        ("version".into(), DOCUMENT_VERSION.into()),
292    ];
293
294    let enc = protocol::encrypt_request(&parameters, rsa_modulus, rsa_exponent)?;
295    let body = protocol::build_post_body(&enc.s, &enc.d);
296
297    let mut last_error: Option<String> = None;
298    for url in bootstrap_urls {
299        match try_fetch(http_client, url, &body, &enc.aes_key, &enc.aes_iv).await {
300            Ok(xml_text) => return Ok(xml_text),
301            Err(e) => {
302                last_error = Some(e.to_string());
303                continue;
304            }
305        }
306    }
307
308    Err(Error::AllServersFailed(
309        last_error.unwrap_or_else(|| "unknown error".into()),
310    ))
311}
312
313// Attempt a single POST to a bootstrap URL and decrypt the response.
314async fn try_fetch(
315    http_client: &reqwest::Client,
316    url: &str,
317    body: &str,
318    aes_key: &[u8],
319    aes_iv: &[u8],
320) -> Result<String> {
321    let resp = http_client
322        .post(url)
323        .header("Content-Type", "application/x-www-form-urlencoded")
324        .body(body.to_string())
325        .send()
326        .await?;
327
328    if resp.status() != reqwest::StatusCode::OK {
329        return Err(Error::HttpStatus(resp.status().as_u16()));
330    }
331
332    let encrypted_response = resp.bytes().await?;
333
334    let xml_bytes = protocol::decrypt_response(&encrypted_response, aes_key, aes_iv)?;
335    let xml_text = std::str::from_utf8(&xml_bytes)?;
336
337    Ok(xml_text.to_string())
338}
