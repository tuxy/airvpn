1//! Error types for the AirVPN library.
2//!
3//! All fallible operations return [`Result<T, Error>`]. The [`Error`] enum
4//! covers HTTP failures, XML/JSON parsing, encryption/decryption issues,
5//! WireGuard config problems, and API-level errors returned by AirVPN.
6
7use base64::DecodeError;
8
9/// All errors that can be produced by this library.
10#[derive(Debug, thiserror::Error)]
11pub enum Error {
12    /// An HTTP request failed (network error, timeout, etc.).
13    #[error("HTTP request failed: {0}")]
14    Http(String),
15
16    /// The server returned a non-200 HTTP status code.
17    #[error("HTTP status {0}")]
18    HttpStatus(u16),
19
20    /// XML parsing failed.
21    #[error("XML parse error: {0}")]
22    Xml(#[from] roxmltree::Error),
23
24    /// JSON deserialization failed.
25    #[error("JSON error: {0}")]
26    Json(#[from] serde_json::Error),
27
28    /// The AirVPN server returned an error message in its XML response.
29    #[error("server returned error: {0}")]
30    Server(String),
31
32    /// Every bootstrap URL was tried and all failed.
33    #[error("all bootstrap servers failed: {0}")]
34    AllServersFailed(String),
35
36    /// RSA or AES encryption failed.
37    #[error("encryption error: {0}")]
38    Encryption(String),
39
40    /// AES decryption or PKCS7 unpadding failed.
41    #[error("decryption error: {0}")]
42    Decryption(String),
43
44    /// An RSA key could not be constructed from the given modulus/exponent.
45    #[error("invalid key: {0}")]
46    InvalidKey(String),
47
48    /// A WireGuard configuration could not be extracted from the user XML.
49    #[error("wireguard config error: {0}")]
50    Wireguard(String),
51
52    /// The public status API returned a JSON error payload.
53    #[error("API error: {0}")]
54    Api(String),
55
56    /// [`AirVPNBuilder::build`] was called without setting credentials.
57    ///
58    /// [`AirVPNBuilder::build`]: crate::AirVPNBuilder::build
59    #[error("missing credentials")]
60    MissingCredentials,
61
62    /// A byte slice that was expected to be valid UTF-8 was not.
63    #[error("UTF-8 error: {0}")]
64    Utf8(#[from] std::str::Utf8Error),
65
66    /// Base64 decoding failed.
67    #[error("base64 decode error: {0}")]
68    Base64(#[from] DecodeError),
69}
70
71/// A specialised `Result` type alias for this library.
72pub type Result<T> = std::result::Result<T, Error>;
73
74impl From<reqwest::Error> for Error {
75    fn from(e: reqwest::Error) -> Self {
76        Error::Http(e.to_string())
77    }
78}
